/**
 * Send a request to fetch user activity
 * @param { String } id
 * @returns { Promise }
 */

export function getActivity(id) {
  return fetch(`http://localhost:3000/user/${id}/activity`)
    .then((response) => response.json())
    .then((result) => {
      // console.log(`Activity:`);
      // console.log(result);
      return result;
    })
    .catch((error) => {
      console.log(`Error Activity: ${error}`);
    });
}
